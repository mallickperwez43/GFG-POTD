class Solution
{
    public:
    //Function to find all elements in array that appear more than n/k times.
    int countOccurence(int arr[], int n, int k) {
        // Your code here
        unordered_map<int,int> freq;
        for(int i = 0 ; i < n ; i++){
            freq[arr[i]]++;
        }
        int check = n / k;
        int count = 0;
        for(auto it : freq){
            if(it.second > check){
                count++;
            }
        }
        return count;
    }
};
